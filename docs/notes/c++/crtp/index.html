    模板参数#

    template <typename T> class Printable {
public:
    // 显示声明参数 可以不使用 static_cast 
  void print(this const auto &self)
  {
    std::println("{}", self.to_string());
  }
};

    为奇异模板添加concept约束

    // concept
concept Print = requires {
  { std::declval<const T &>().to_string() } -> std::same_as<std::string>;
};
template <typename T> class Printable {
public:
  void print(this const auto &self)
      // 确保需要Printable功能的类存在to_string函数
    requires Print<T>
  {
    std::println("{}", self.to_string());
  }
};

    奇异模板提供了一种功能聚合的功能

    // CRTP: 模板模板参数
template <template <typename> class... Capabilities>
class Student : public Capabilities<Student<Capabilities...>>... {
public:
  std::string name{"liuda"};
  int age{10};
  std::string to_string(this const Student &self) {
    return std::format("name: {},age: {}", self.name, self.age);
  };
};
//这样声明的类可以组合功能

  Student<Printable> student1{};
