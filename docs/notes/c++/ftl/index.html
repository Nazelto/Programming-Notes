    %运算符#

    %需要明确导入using ftl::operator%类似Functor功能。

    void functor() {
  using ftl::operator%;
  ftl::maybe<int> just = ftl::just(5);
  auto doubles = [](int x) { return x * 2; };
  doubles % just;
}

    »= 运算符#

    >>=运算符与haskell一致

    // 这里的std::list 是一个monad
void flat(std::list<int> &list) {
  using ftl::operator>>=;
  auto mapper = [](int x) { return std::list<int>{x - 1, x, x + 1}; };
  list = list >>= mapper;
}
