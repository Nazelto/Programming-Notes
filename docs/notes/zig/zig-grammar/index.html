    zig 语法#

    流程控制#

    if#

    if可用来解包可选类型(?) 与 错误类型(!)

    const optional:?i32 = null;
if (optional) |number:i32| {
    //something todo
}
const error: anyerror!i32 = 0;
if (error) |number:i32| {
    //something todo
}

    高级类型#

    指针#

    单项指针#

    • 单项指针指向单个元素。

    • 单项指针的类型为 *T，其中 T 是所指向的数据类型。解引用操作使用 ptr.*。

    // 代码来自 zig 语言圣经
const print = @import("std").debug.print;

pub fn main() !void {
    var integer: i16 = 666;
    const ptr = &integer;
    ptr.* = ptr.* + 1;

    print("{}\n", .{integer});
}

    多项指针#

    • 多项指针指向一个或多个连续的元素，但其数量在编译期是未知的。

    TIP
    多项指针的类型为 [*]T，其中 T 是所指向的数据类型，且 T 的大小必须是明确的（这意味着它不能是 anyopaque 或其他不透明类型）

    const print = @import("std").debug.print;

pub fn main() !void {
    const array = [_]i32{ 1, 2, 3, 4 };
    const ptr: [*]const i32 = &array;

    print("第一个元素：{}\n", .{ptr[0]});
}

    多个指针的区别#

    [4] const u8 // 该类型表示一个长度为4的数组，里面的类型
[] const u8 // 这表示是一个 内部类型为 const u8 的切片
*[4] const u8 // 表示一个指针，该指针的地址存储一个长度为4，元素为const u8的数组
*[] const u8 //这表示指针地址存储着一个切片
[*] const u8 // 表示一个多项指针 元素为const u8 但是长度未知

    切片与数组#

    切片#

    切片本质上是一个胖指针，内部有一个[*] T的多项指针与切片长度。

    const array: [_] i32 = [_]i32{1,2,3,4};
const slice:[] i32 = array[0..4];

    联合类型#

    union与tag union#
    union {
  int:i32,
  float:f32
};

    union的缺陷是他不知道自己存储的是什么类型的数据。而tag union就是来解决这个问题的。

    union(enum) {
    a:i32,
    b:f32,
};
