    快速排序#

    pub fn quick_sort(seq: &[i32]) -> Vec<i32> {
    if let [pivot, rest @ ..] = seq {
        let (left, right): (Vec<i32>, Vec<i32>) = rest.iter().partition(|x| *x < pivot);
        [quick_sort(&left), vec![*pivot], quick_sort(&right)].concat()
    } else {
        vec![]
    }
}

    归并排序#

    pub fn merge_sort(seq: &[i32]) -> Vec<i32> {
    match seq {
        [] | [_] => seq.to_vec(),
        _ => {
            let mid = seq.len() / 2;
            let (left, right) = seq.split_at(mid);
            merge(&merge_sort(left), &merge_sort(right))
        }
    }
}

fn merge(left: &[i32], right: &[i32]) -> Vec<i32> {
    let mut left_iter = left.iter().peekable();
    let mut right_iter = right.iter().peekable();
    let mut result = Vec::with_capacity(left.len() + right.len());

    while let (Some(l), Some(r)) = (left_iter.peek(), right_iter.peek()) {
        if l <= r {
            result.push(*left_iter.next().unwrap());
        } else {
            result.push(*right_iter.next().unwrap());
        }
    }
    result.extend(left_iter);
    result.extend(right_iter);
    result
}

    选择排序#

    pub fn select_sort(seq: &mut [i32]) -> &[i32] {
    (0..seq.len()).for_each(|i| {
        let min_pos = (i..seq.len()).min_by_key(|&j| seq[j]).unwrap_or(i);
        seq.swap(i, min_pos);
    });
    seq
}

    冒泡排序#

    pub fn bubbled_sort(seq: &mut [i32]) -> &[i32] {
    for i in 1..seq.len() {
        for j in (i..seq.len()).rev() {
            if seq[j] < seq[j - 1] {
                seq.swap(j, j - 1);
            }
        }
    }
    seq
}

    滑动窗口#

    pub fn windows<'a>(seq: &'a [u8], pattern: &'a [u8]) -> Option<&'a [u8]> {
    use std::ops::ControlFlow;

    let pattern_len = pattern.len();

    for wsize in pattern_len..=seq.len() {
        let result = seq.windows(wsize).try_for_each(|chunk| {
            let all_found = pattern.iter().all(|elem| chunk.contains(elem));

            if all_found {
                ControlFlow::Break(chunk)
            } else {
                ControlFlow::Continue(())
            }
        });

        if let ControlFlow::Break(chunk) = result {
            return Some(chunk);
        }
    }

    None
}
