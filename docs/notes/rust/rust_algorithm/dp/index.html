pub fn longest_palindrome(s: String) -> String {
    let len = s.len();
    if len == 0 {
        return s; // 直接处理边界
    }

    let s_ref = s.as_bytes();
    let mut array: Vec<Vec<bool>> = vec![vec![false; len]; len]; // 简化数组

    // 初始化：使用 map 处理
    for i in 0..len {
        array[i][i] = true; // 可以用迭代器，但这里保持简单
    }

    // 函数式处理 DP 填充：用高阶函数和 match
    fn fill_dp(array: &mut Vec<Vec<bool>>, s_ref: &[u8], len: usize) {
        for L in 2..=len {
            // 按长度迭代
            (0..=len - L).for_each(|i| {
                let j = i + L - 1;
                match (s_ref.get(i), s_ref.get(j)) {
                    // 用 match 处理边界和相等
                    (Some(&a), Some(&b)) if a == b => {
                        // 模式匹配检查相等
                        match L {
                            2 => array[i][j] = true,                                     // 长度为 2
                            _ if i < j - 1 && array[i + 1][j - 1] => array[i][j] = true, // 子问题
                            _ => array[i][j] = false,
                        }
                    }
                    _ => array[i][j] = false, // 边界或不相等
                }
            });
        }
    }

    fill_dp(&mut array, s_ref, len); // 调用函数

    // 查找最长子串：用 fold 减少 if
    let (start, max_length) = (0..len).fold((0, 1), |(start, max_len), i| {
        let candidates: Vec<(usize, usize)> = (i..len)
            .filter(|&j| array[i][j])
            .map(|j| (i, j - i + 1))
            .filter(|&(_, length)| length > max_len)
            .collect();
        if let Some(&(new_start, new_length)) = candidates.iter().max_by_key(|&&(_, length)| length)
        {
            // 如果有候选，选择最大的
            (new_start, new_length)
        } else {
            (start, max_len) // 无候选，保持原状态
        }
    });

    s[start..start + max_length].to_string() // 返回结果
}
