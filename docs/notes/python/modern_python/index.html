    装饰器#

    # @wrap 就是装饰器
@wrap
def add(x:int,y:int) -> int:

    1. 装饰器可以简单理解为在运行函数之前执行与函数执行结束之后执行一些特定的操作。

    2. 装饰器也是函数。

    3. 被装饰器修饰的函数将会在装饰器内部执行。

    自定义装饰器#
    from typing import Callable,Any
from time import perf_counter,sleep
# 计时装饰器
def time(func:Callable) -> Callable:
    def wrapper(*args,**kargs) -> Any:
        print(f'Timing "{func.__name__}"...')
        start:float = perf_counter()
        result:Any = func(*args,**kargs)
        end:float = perf_counter()
        print(f'"{func.__name__}" took: {end - start: .4f}s')
        return result
    return wrapper
# 函数式写法
def time(func:Callable) -> Callable:
    @wraps(func)
    def wrapper(*args,**kargs) -> Any:
