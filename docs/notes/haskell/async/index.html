    Haskell Async 库详解#

    概述#

    Async 是 Haskell 中用于并发编程的核心库，提供了类型安全的异步操作管理，允许开发者以声明式风格编写并发代码。

    核心概念#

    Async 类型#

    data Async a = Async { 
    asyncThreadId :: ThreadId,
    -- 其他内部字段
}

    表示一个正在进行的异步计算，最终会产生一个 a 类型的值。

    主要功能#

    启动异步任务#

    -- 基本异步操作
async :: IO a -> IO (Async a)

-- 带资源管理的异步
withAsync :: IO a -> (Async a -> IO b) -> IO b

-- 在特定CPU核心上运行
asyncOn :: Int -> IO a -> IO (Async a)

    等待结果#

    -- 阻塞等待结果
wait :: Async a -> IO a

-- 等待并捕获异常
waitCatch :: Async a -> IO (Either SomeException a)

-- 等待任意一个完成
waitAny :: [Async a] -> IO (Async a, [Async a])

-- 等待全部完成
waitBoth :: Async a -> Async b -> IO (a, b)

    组合操作#

    -- 竞争：返回先完成的结果
race :: IO a -> IO b -> IO (Either a b)

-- 并发执行两个操作
concurrently :: IO a -> IO b -> IO (a, b)

-- 并发执行多个相同操作
replicateConcurrently :: Int -> IO a -> IO [a]

    遍历并发#

    -- 并发遍历列表
mapConcurrently :: Traversable t => (a -> IO b) -> t a -> IO (t b)

-- 并发循环
forConcurrently :: Traversable t => t a -> (a -> IO b) -> IO (t b)

-- 限制并发数
mapConcurrentlyN :: Int -> (a -> IO b) -> [a] -> IO [b]

    实用示例#

    基本使用#

    import Control.Concurrent.Async
import Control.Concurrent (threadDelay)

main :: IO ()
main = do
    -- 启动两个异步任务
    a1 <- async $ do
        threadDelay 1000000
        return "Task 1 done"

    a2 <- async $ do
        threadDelay 2000000
        return "Task 2 done"

    -- 等待两个任务完成
    r1 <- wait a1
    r2 <- wait a2

    putStrLn $ r1 ++ ", " ++ r2

    withAsync 自动清理#

    main :: IO ()
main = do
    -- withAsync 确保资源被正确释放
    withAsync (longRunningTask "Task1") $ \a1 ->
        withAsync (longRunningTask "Task2") $ \a2 -> do
            r1 <- wait a1
            r2 <- wait a2
            print (r1, r2)
  where
    longRunningTask name = do
        threadDelay 500000
        return name

    并发 Web 请求#

    import Control.Concurrent.Async
import Network.HTTP.Simple
import qualified Data.ByteString.Char8 as B

fetchUrls :: [String] -> IO [B.ByteString]
fetchUrls urls = do
    -- 并发获取多个URL
    mapConcurrently fetch urls
  where
    fetch url = do
        response <- httpBS (fromString url)
        return $ getResponseBody response

    超时处理#

    import Control.Concurrent.Async
import Control.Concurrent (threadDelay)

withTimeout :: Int -> IO a -> IO (Maybe a)
withTimeout micros action = do
    result <- race (threadDelay micros >> return Nothing)
                   (fmap Just action)
    case result of
        Left _ -> return Nothing
        Right x -> return x

main :: IO ()
main = do
    result <- withTimeout 500000 $ do
        threadDelay 1000000  -- 这会超时
        return "Success"
    case result of
        Just x -> putStrLn x
        Nothing -> putStrLn "Timeout!"

    高级特性#

    异常处理#

    safeAsync :: IO a -> IO (Either SomeException a)
safeAsync action = do
    a <- async action
    waitCatch a

    取消操作#

    cancelWith :: Async a -> Async b -> IO ()
-- 当第一个async完成或异常时，取消第二个

    链接异步任务#

    link :: Async a -> Async b -> IO ()
-- 链接两个异步任务，一个失败时另一个也被取消

    STM 集成#

    poll :: Async a -> STM (Maybe (Either SomeException a))
-- 非阻塞检查异步任务状态

    最佳实践#

    1. 资源管理#

    总是使用 withAsync 而非 async + cancel 来确保资源清理。

    2. 限制并发数#

    避免创建过多线程，使用 mapConcurrentlyN 或连接池。

    3. 错误处理策略#

    -- 收集所有结果，包括失败的
allResults <- mapConcurrently (try @SomeException) actions

    4. 性能监控#

    import Control.Concurrent.Async.Timer

timedAsync :: IO a -> IO (Async a, Async ())
timedAsync action = do
    (a, timer) <- asyncTimed action
    _ <- async $ do
        elapsed <- wait timer
        putStrLn $ "Task took: " ++ show elapsed
    return a

    常见模式#

    扇出/扇入模式#

    fanOutFanIn :: [IO a] -> (a -> IO b) -> IO [b]
fanOutFanIn producers consumer = do
    -- 扇出：并发执行生产者
    inputs <- mapConcurrently id producers
    -- 扇入：并发处理结果
    mapConcurrently consumer inputs

    管道并行#

    pipeline :: [a] -> (a -> IO b) -> (b -> IO c) -> IO [c]
pipeline inputs stage1 stage2 = do
    -- 第一阶段并发
    intermediate <- mapConcurrently stage1 inputs
    -- 第二阶段并发
    mapConcurrently stage2 intermediate

    注意事项#

    1. 内存使用：大量并发任务可能消耗大量内存
    2. 异常传播：未捕获的异常会终止整个程序
    3. 线程安全：确保共享数据的同步访问
    4. 死锁避免：注意异步任务间的依赖关系

    相关扩展库#

    • async-pool：连接池管理
    • async-timer：带计时的异步操作
    • stm-conduit：与conduit集成
    • thread-local-storage：线程局部变量管理

    安装#

    # 使用 cabal
cabal install async

# 使用 stack
stack install async

    性能特点#

    特性 说明
    轻量级线程 Haskell 的轻量级线程支持数千个并发任务
    真正的并行 在多核CPU上可以真正并行执行
    非阻塞I/O 自动利用系统的非阻塞I/O机制
    低开销 上下文切换开销极小

    这个库通过提供高级抽象，使得Haskell中的并发编程变得更加直观和安全，是构建高性能并发应用的强大工具。
