    flake-parts#

    flake-parts模块化#

    {
  inputs.flake-parts.url = "github:hercules-ci/flake-parts";
  
  outputs = inputs@{ flake-parts, ... }:
    flake-parts.lib.mkFlake { inherit inputs; } {
      # 每个配置块都是独立的模块
      systems = [ "x86_64-linux" "aarch64-darwin" ];
      
      perSystem = { config, pkgs, ... }: {
        packages.default = pkgs.hello;
        devShells.default = pkgs.mkShell { };
      };
    };
}
    # flake.nix
{
  outputs = inputs@{ flake-parts, ... }:
    flake-parts.lib.mkFlake { inherit inputs; } {
      imports = [
        ./packages.nix      # 包定义模块
        ./devshells.nix     # 开发环境模块
        ./nixos.nix         # NixOS 配置模块
        ./checks.nix        # 测试检查模块
      ];
    };
}

# packages.nix
{ ... }: {
  perSystem = { pkgs, ... }: {
    packages = {
      app1 = pkgs.callPackage ./app1 { };
      app2 = pkgs.callPackage ./app2 { };
    };
  };
}

# devshells.nix
{ ... }: {
  perSystem = { pkgs, ... }: {
    devShells.default = pkgs.mkShell {
      packages = [ pkgs.git pkgs.nodejs ];
    };
  };
}

    flake创建包#

    {
  description = "Functional Template Library for C++";
  inputs = {
    nixpkgs.url = "github:nixos/nixpkgs/nixos-25.11";
    flake-parts.url = "github:hercules-ci/flake-parts";
  };
  outputs =
    inputs@{
      self,
      nixpkgs,
      flake-parts,
      ...
    }:
    flake-parts.lib.mkFlake { inherit inputs; } {
      systems = [
        "x86_64-linux"
        "aarch64-linux"
        "x86_64-darwin"
        "aarch64-darwin"
      ];
      perSystem =
        {
          config,
          self',
          inputs',
          pkgs,
          system,
          ...
        }:
        {
          packages = {
            default = pkgs.callPackage ./ftl.nix { };
          };
        };
      flake = {
        overlays.default = final: prev: {
          ftl = final.callPackage ./ftl.nix { };
        };
      };
    };
}

    使用flake-parts打包c++ ftl库

    overlays 覆盖你的pkgs将你的ftl加入nixpkgs 可以像下面一样使用

    let
    pkgs = import nixpkgs {
        # overlays 这里必须是list 为了组合多了overlays
        overlays = [ftl.overlays.default];
        };
in
buildInputs = with pkgs; [ftl];
